# Function to create a Directed graph
def get_graph(x):
    G = nx.DiGraph()
    G=nx.from_pandas_edgelist(x,'FROM', 'TO', create_using=nx.DiGraph())
    return G

pos = nx.circular_layout(G)

d = dict(G.degree)

nx.draw_networkx_edge_labels(G,pos,edge_labels=labels,font_color='black', font_size=15)
